## Generating Language-Specific Code from .proto

python -m grpc_tools.protoc -I=proto --python_out=. --grpc_python_out=. proto/service.proto

protoc --java_out=src/main/java --grpc-java_out=src/main/java --plugin=protoc-gen-grpc-java=$(which protoc-gen-grpc-java) --proto_path=proto/service.proto

protoc --java_out=src/main/java \
       --grpc-java_out=src/main/java \
       --plugin=protoc-gen-grpc-java=$(which protoc-gen-grpc-java) \
       --proto_path=proto \
       proto/service.proto
